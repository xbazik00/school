\item Ak pre prvok $y \in Y$ existuje jedin\'e tak\'e $x$, \v ze $f(x)=y$, tak obrazu $y$ prirad\'ime rovnak\'u pr\'islu\v snos\v t do mno\v ziny $B$, ako pr\'islu\v snos\v t vzoru $x$ do mno\v ziny $A$, teda $\mu_B(y)=\mu_A(x).$

\item Ak prvok $y$ nem\'a vzor, tak $\mu_B(y)=0.$
\end{itemize}
\end{itemize} 


Uva\v zujme zobrazenie $f:X \to Y; y = f(x)$ tak\'e, \v ze mno\v zina vzorov $f^{-1}\left(\{y\}\right)$ je v\v zdy kone\v cn\'a. Pre fuzzy mno\v zinu $A \in F(X)$ vytvorme jej horizont\'alnu reprezent\'aciu 
$\left( A^{(\alpha)}\right)_{\alpha \in \left( 0,1\right\rangle}.$ V\v setky $\alpha-$ rezy $A^{(\alpha)}$ s\'u ostr\'e mno\v ziny a preto ich obraz v zobrazen\'i je
$$f\left(A^{(\alpha)}\right)=\{y \in Y; \exists x \in A^{(\alpha)}; y =f(x)\}.$$
Uk\'a\v zte, \v ze syst\'em 
$f\left(A^{(\alpha)}\right)_{\alpha \in \left( 0,1\right\rangle}$ je za na\v sich predpokladov syst\'em $\alpha-$rezov nejakej fuzzy mno\v ziny.

{\bf Pr\'iklady}
\begin{itemize}
\item Nech $A$ je fuzzy podmno\v zina univerza cel\'ych \v c\'isel $Z,$
$$A=\{0.2/-1,0.3/0,0.6/1,0.4/2\}.$$
Ur\v te fuzzy mno\v zinu $A^2.$

\item Nech $A$ je fuzzy podmno\v zina univerza re\'alnych \v c\'isel $R,$
$$\mu_A=\begin{cases}
x & \mbox{ak } x \in \langle 0,1 \rangle\\
2-x & \mbox{ak } x \in \left( 1,2 \right\rangle\\
0 & \mbox{pre ostatn\'e  } x \in R.
\end{cases}$$
Ur\v cte hodnotu roz\v s\'irenia exponenci\'alnej funkcie $\exp(A)=e^{A}.$

\end{itemize} 








\centerline{\Large\sc Princ\'ip roz\v s\'irenia pre zobrazenia viac premenn\'ych }
\bigskip

Nech $f: X\times Y \to U, u = f(x,y). $ Potom $f(A,B)=\{f(x,y); x \in A, y \in B\}=f(A\times B).$ 

Nech $\overline{f}$ pracuje s fuzzy podmno\v zinami. 
Potom fuzzy podmno\v zin\'am $A, B$ prirad\'ime fuzzy podmno\v zinu $C$, pre ktor\'u plat\'i:
$$\mu_C(u)=\sup\limits_{u=f(x,y)}\mu_{A\times B}(x,y)$$
teda
$$\mu_C(u)=\sup\limits_{u=f(x,y)}\min(\mu_{A}(x),\mu_B(y))$$

To znamen\'a, \v ze ak m\'a prvok $u \in U$ viac vzorov, prirad\'ime mu tak\'y stupe\v n pr\'islu\v snosti do obrazovej  mno\v ziny, ktor\'y je rovn\'y maxm\'alnemu stup\v nu pr\'islu\v snosti t\'ych dvoj\'ic z $A \times B$, ktor\'e sa na\v n zobrazia.

\Large\sc \v Standardn\'e oper\'acie.} 


{\bf Defin\'icia.}
Nech $X$ je z\'akladn\'y priestor $A, B \in F(X)$.
\begin{itemize}
\item {\bf \v Standardn\'y prienik fuzzy mno\v z\'in} $A, B$ je fuzzy mno\v zina  $A \cap B \in F(X) $ s funkciou pr\'islu\v snosti:
$$\mu_{A \cap B}(x)=\min\left(\mu_A(x), \mu_B(x)\right).$$
\item {\bf \v Standardn\'e zjednotenie fuzzy mno\v z\'in} $A, B$ je fuzzy mno\v zina  $A \cup B \in F(X) $ s funkciou pr\'islu\v snosti:
$$\mu_{A \cup B}(x)=\max\left(\mu_A(x), \mu_B(x)\right).$$
\item {\bf \v Standardn\'ym doplnkom fuzzy mno\v ziny} $A$ je fuzzy mno\v zina  
$\bar{A} \in F(X) $ s funkciou pr\'islu\v snosti:
$$\mu_{\bar{A}}(x)=1-\mu_A(x).$$

\end{itemize}

\bigskip

{\Large\sc Pr\'iklady.}

\bigskip

\begin{itemize}
\item Nech $X=\left\langle 0, \infty \right)$ a nech $A, B \in F(X)$
$$A=\{0.2/-1,0.4/0,0.7/1,0.3/2\}, ~~~ B=\{0.5/-1,0.4/0, 0.4/1,0.8/2\}.$$
Ur\v cte $A \cap B, A \cup B, \bar{A}$
\item Nech $X=\{-1,0,1,2\}$ a nech $A, B \in F(X)$
$$\mu_A= \begin{cases} \frac 12(2-x) & x \in \left\langle 0,2 \right\rangle\\
0 & inde, \end{cases}
~~~ \mu_B = \begin{cases} \frac 12x & x \in \left\langle 0,2 \right\rangle\\
(3-x) & x \in \left( 2,3 \right\rangle\\
0 & inde. \end{cases}$$
Ur\v cte $A \cap B, A \cup B, \bar{A}, \bar{B}$
\end{itemize}

{\Large\sc  Vlastnosti \v standardn\'ych oper\'aci\'i s fuzzy mno\v zinami}

Vieme, \v ze pre klasick\'e mno\v ziny platia nasleduj\'uce vlastnosti:

\begin{itemize}
\item Komutat\'ivnos\v t \begin{itemize}
\item $A \cap B=B\cap A$
\item $A \cup B=B\cup A$

\end{itemize}

\item Asociat\'ivnos\v t
\begin{itemize}
\item $A \cap \left(B \cap C\right)=\left( A \cap B\right)\cap C$
\item $A \cup \left(B \cup C\right)=\left( A \cup B\right)\cup C$
\end{itemize}

\item Distribut\'ivnos\v t
\begin{itemize}
\item $A \cap \left(B \cup C\right)=\left( A \cap B\right)\cup \left( A\cap C\right)$
\item $A \cup \left(B \cap C\right)=\left( A \cup B\right)\cap \left( A\cup C\right)$
\end{itemize}

\item Idempotencia
\begin{itemize}
\item $A \cap A= A$
\item $A \cup A= A$

\end{itemize}
\item Absorbcia
\begin{itemize}
\item $A \cap \left(A \cup B\right)= A$
\item $A \cup \left(A \cap B\right)= A$

\end{itemize}
\item Absorbcia $X$ a $\emptyset$
\begin{itemize}
\item $A \cap \emptyset= \emptyset$
\item $A \cup X= X$

\end{itemize}

\item Identita
\begin{itemize}
\item $A \cap X= A$
\item $A \cup \emptyset= A$

\end{itemize}

\item Involut\'ivnos\v t
\begin{itemize}
\item $\bar{\bar{A}}=A$
\end{itemize}
\item Z\'akon sporu
\begin{itemize}
\item $A \cap \bar{A}= \emptyset$
\end{itemize}
\item Z\'akon vyl\'u\v cenia tretieho
\begin{itemize}
\item $A \cup \bar{A}= X$
\end{itemize}
\item De Morganove z\'akony
\begin{itemize}
\item $\overline{A\cap B} = \bar{A} \cup \bar{B}$
\item $\overline{A \cup B} = \bar{A} \cap \bar{B}$

\end{itemize}

\end{itemize}

{\Large\sc $\alpha$ - rezy fuzzy mno\v z\'in}

\bigskip

{\bf Defin\'icia.} Nech $A$ je fuzzy podmno\v zina univerza $X$ a nech $\alpha \in \left\langle 0,1 \right\rangle$. $\alpha$- rezom fuzzy mno\v ziny $A$ sa naz\'yva mno\v zina 
$$A^{(\alpha)}=\{x \in X; \mu_A(x)\geq \alpha\}.$$
Zrejme $Ker(A)=A^{(1)}, A^{(0)}=X.$
Ka\v zdej fuzzy mno\v zine vieme priradi\v t syst\'em $\alpha-$ rezov, \v ci\v ze syst\'em ost\'ych mno\v z\'in. Teda fuzzy mno\v zina sa d\'a rozlo\v zi\v t na syst\'em ostr\'ych mno\v z\'in-dekompoz\'icia fuzzy mno\v ziny.

\bigskip

{\bf Reprezent\'acia fuzzy mno\v z\'in:}
\begin{itemize}
\item {\bf horizont\'alna} - mno\v zina je zadan\'a pomocou $\alpha-$ rezov
\item {\bf vertik\'alna} - mno\v zina je zadan\'a pomocou funkcie pr\'islu\v snosti
\end{itemize}

\bigskip

{\Large\sc Pr\'iklady.}

\bigskip
\begin{itemize}
\item Nech $X=\{1,2,3,4\}, A=\{0.2/1,0.4/2,0.8/3,0.5/4\}.$ Ur\v cte $\alpha-$ rezy fuzzy podmno\v ziny $A.$
\item Nech $X=R$
$$\mu_A = \begin{cases} \frac 12(x-1) & x \in \left\langle 1,3 \right\rangle\\
(4-x) & x \in \left( 3,4 \right\rangle\\
0 & inde. \end{cases}$$
 Ur\v cte $\alpha-$ rezy fuzzy podmno\v ziny $A.$


\end{itemize}

\newpage

{\bf Dok\'a\v zte nasleduj\'uce tvrdenia:}

\bigskip

{\bf Veta.}
Nech $A, B \in F(X).$ Potom
\begin{itemize}
\item Syst\'em $\alpha-$rezov $\left(A^{(\alpha)}\right)_{a \in \left(0,1\right\rangle}$ je nerast\'uci, t.j.,
$$\forall \alpha_1, \alpha_2 \in \left(0,1\right\rangle; \alpha_1 <\alpha_2 \Rightarrow A^{(\alpha_2)} \subseteq A^{(\alpha_1)}$$
\item $Supp(A)=\bigcup\limits_{\alpha \in \left(0,1\right\rangle} A^{(\alpha)}$
\item $A \subseteq B \iff A^{(\alpha)} \subseteq B^{(\alpha)} \forall \alpha \in \left(0,1\right\rangle$
\item $A = B \iff A^{(\alpha)} = B^{(\alpha)} \forall \alpha \in \left(0,1\right\rangle$

\end{itemize}

\bigskip

{\bf Veta.}
Nech $A, B \in F(X).$ Potom pre ka\v zd\'e $\alpha \in \left(0,1\right\rangle$ plat\'i:
\begin{itemize}
\item $\left( A \cap B\right)^{(\alpha)}= A^{(\alpha)} \cap B^{(\alpha)}$
\item $\left( A \cup B\right)^{(\alpha)}= A^{(\alpha)} \cup B^{(\alpha)}$
\end{itemize}

\bigskip

{\bf Veta.}
Syst\'em ostr\'ych mno\v z\'in $\left(E_{\alpha}\right)_{\alpha \in \left(0,1\right\rangle,} E_{\alpha} \subseteq X$ je syst\'emom $\alpha-$rezov nejakej fuzzy podmno\v ziny $A \in F(X)$ pr\'ave vtedy, ak
\begin{itemize}
\item syst\'em $\left(E_{\alpha}\right)_{\alpha \in \left(0,1\right\rangle}$ je nerast\'uci 
\item $\forall \alpha \in \left(0,1\right\rangle; E_\alpha = \bigcap\limits_{\beta < \alpha}E_\beta.$
\end{itemize}

\bigskip

{\bf Veta.}
Nech $\left(A_{\alpha}\right)_{\alpha \in \left(0,1\right\rangle,}$ je syst\'em $\alpha-$rezov fuzzy mno\v ziny $A \in F(X).$ Potom pre ka\v zd\'e $x \in X$ plat\'i:
$$\mu_A(x)= \sup\limits_{\alpha \in \left(0,1\right\rangle}\min\left(\alpha, \chi_{A^{(\alpha)}}(x)\right).$$

\begin{itemize}
\item kart\'ezsky s\'u\v cin
$$ A \times B =\{[x,y]; x \in A \wedge y \in B\}$$
$$[x,y] \in A \times B \iff x \in A \wedge y \in B$$
$$[x,y] \not \in A \times B \iff x \not \in A \vee y \not \in B$$
Vlastnosti:
\begin{itemize}
\item  $$(A \cup B)\times C =(A \times C) \cup (B \times C)$$
\item  $$(A \cap B)\times C =(A \times C) \cap (B \times C)$$
\item  $$A \times (B \cup C) =(A \times B) \cup (A \times C)$$
\item  $$A \times (B\cap C) =(A \times B) \cap (A \times C)$$

\end{itemize}

\end{itemize}
\end{frame}

\begin{frame}
\begin{remark}[Permut\'acie bez opakovania]
Permut\'aciou bez opakovania z $n$-prvkovej mno\v ziny $M$ naz\'yvame ka\v
zd\'u usp.
$n-$ticu navz\'ajom r\^ oznych prvkov, vytvoren\'u z prvkov mno¾iny $M.$ Poèet
v¹etk\'ych permut\'acií je 
$$P(n) = 1 \cdot 2 \cdot 3 \dots \cdot n = n!$$
\end{remark}
\end{frame}





\begin{frame}
\begin{remark}[Vari\'acie bez opakovania]
Vari\'aciou bez opakovania $k-$tej triedy  z $n$-prvkovej mno\v ziny $M$ naz\'yvame ka\v
zd\'u usp.
$k-$ticu navz\'ajom r\^oznych prvkov, vytvoren\'u z prvkov mno¾iny $M.$ Poèet
v¹etk\'ych vari\'acií je 
$$V_k(n) = n \cdot (n-1) \cdot (n-2) \dots \cdot (n-k+1) = \frac{n!}{(n-k)!}
= {n \choose k} \cdot k!$$
\end{remark}
\end{frame}




\begin{frame}
\begin{remark}[Kombin\'acie bez opakovania]
Kombin\'aciou bez opakovania $k-$tej triedy  z $n$-prvkovej mno\v ziny $M$ naz\'yvame ka\v
zd\'u 
$k-$prvkov\'u podmno\v zinu  vytvoren\'u z prvkov mno¾iny $M.$  Poèet
v¹etk\'ych kombin\'acií je 
$$C_k(n) = \frac{ n \cdot (n-1) \cdot (n-2) \dots \cdot (n-k+1)}{k \cdot
(k-1) \dots 2 \cdot 1} = \frac{n!}{(n-k)!k!}
= {n \choose k} $$
\end{remark}
\end{frame}

\begin{frame}
\begin{remark}[Vari\'acie s opakovan\'im]
Vari\'aciou s opakovan\'im $k-$tej triedy  z $n$-prvkovej mno\v ziny $M$ naz\'yvame ka\v
zd\'u usp.
$k-$ticu vytvoren\'u z prvkov mno¾iny $M$ tak, \v ze v tejto $k-$tici  sa
ka\v zd\'y prvok m\^ o¾e vyskytn\'u\v t a¾ $k-$kr\'at. Poèet
v¹etk\'ych vari\'acií je 
$$V_k'(n) = n^k.$$
\end{remark}
\end{frame}
\begin{frame}
\begin{remark}[Kombin\'acie s opakovan\'im]
Kombin\'aciou s opakovan\'im $k-$tej triedy  z $n$-prvkovej mno\v ziny $M$ naz\'yvame ka\v
zd\'u skupinu o 
$k-$prvkoch  vytvoren\'u z prvkov mno¾iny $M$ tak, \v ze v tejto skupine  sa
ka\v zd\'y prvok m\^ o¾e vyskytn\'ut a¾ $k-$kr\'at.  Poèet
v¹etk\'ych kombin\'acií je 
$$C_k'(n) = {{n-1+k} \choose k} $$
\end{remark}
\end{frame}

\begin{frame}
\begin{remark}[Permut\'acie s opakovan\'im]
Permut\'aciou s opakovan\'im z $n$-prvkovej mno\v ziny $M=\{a_1, a_2, ...
a_p\}$ s opakovan\'im prvku $a_i$ pr\'ave $k_i$-kr\'at naz\'yv\'ame ka\v
zd\'u usp.
$n-$ticu navz\'ajom r\^ oznych prvkov, vytvoren\'u zo v\v estk\'ych prvkov mno¾iny $M,$
\v ze sa prvek $a_i$ vyskytuje pr\'ave $k_i-$kr\'at ($k_1 + k_2 + ...
+k_p =n$.) Poèet
v¹etk\'ych permut\'acií je 
$$P'(n) = \frac{ n!}{k_1!k_2! ... k_p!}$$
\end{remark}
\end{frame}
